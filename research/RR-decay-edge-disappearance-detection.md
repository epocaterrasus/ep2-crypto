Fit a linear regression to the rolling Sharpe over the last N windows:\n\n$$SR_t = \\alpha + \\beta \\cdot t + \\epsilon_t$$\n\nIf $\\beta < 0$ with statistical significance (p < 0.05), the edge is decaying. The magnitude of $\\beta$ indicates how fast.\n\n**Confidence intervals on rolling Sharpe:** The standard error of the Sharpe ratio (Lo, 2002):\n\n$$SE(SR) \\approx \\sqrt{\\frac{1 + SR^2/2}{w}}$$\n\nThis is essential because a Sharpe of 0.5 calculated over 100 observations has a 95% CI of roughly [0.3, 0.7] — wide enough that noise alone can make it look like decay.\n\n### Python Implementation\n\n```python\n@dataclass\nclass RollingSharpeMonitor:\n    \"\"\"\n    Multi-horizon rolling Sharpe ratio monitor with statistical alerts.\n    \"\"\"\n    annualization_factor: float = 105120.0  # 288 bars/day * 365 days\n    windows: tuple = (2016, 4032, 8640)     # 7d, 14d, 30d in 5-min bars\n    baseline_sharpe: float = 0.0            # from training/backtest\n    \n    # State\n    _returns: list = field(default_factory=list)\n    _alert_history: list = field(default_factory=list)\n    \n    def update(self, return_value: float) -> dict:\n        self._returns.append(return_value)\n        n = len(self._returns)\n        \n        result = {\n            \"sharpe_ratios\": {},\n            \"alert_level\": \"normal\",\n            \"alerts\": [],\n            \"trend\": None,\n        }\n        \n        for w in self.windows:\n            if n < w:\n                continue\n                \n            window_returns = np.array(self._returns[-w:])\n            mean_r = np.mean(window_returns)\n            std_r = np.std(window_returns, ddof=1)\n            \n            if std_r < 1e-12:\n                continue\n            \n            sr = (mean_r / std_r) * np.sqrt(self.annualization_factor)\n            se = np.sqrt((1 + sr**2 / 2) / w)\n            \n            days = w / 288\n            result[\"sharpe_ratios\"][f\"{int(days)}d\"] = {\n                \"sharpe\": sr,\n                \"se\": se,\n                \"ci_lower\": sr - 1.96 * se,\n                \"ci_upper\": sr + 1.96 * se,\n                \"ratio_to_baseline\": sr / self.baseline_sharpe if self.baseline_sharpe != 0 else 0,\n            }\n        \n        # Determine alert level\n        alerts = []\n        sr_data = result[\"sharpe_ratios\"]\n        \n        if \"7d\" in sr_data and sr_data[\"7d\"][\"sharpe\"] < self.baseline_sharpe * 0.5:\n            alerts.append(\"7d Sharpe below 50% of baseline\")\n        if \"14d\" in sr_data and sr_data[\"14d\"][\"sharpe\"] < self.baseline_sharpe * 0.5:\n            alerts.append(\"14d Sharpe below 50% of baseline\")\n        if \"30d\" in sr_data and sr_data[\"30d\"][\"sharpe\"] < 0:\n            alerts.append(\"30d Sharpe is negative\")\n        \n        if \"30d\" in sr_data and sr_data[\"30d\"][\"sharpe\"] < 0:\n            result[\"alert_level\"] = \"critical\"\n        elif \"14d\" in sr_data and sr_data[\"14d\"][\"sharpe\"] < self.baseline_sharpe * 0.5:\n            result[\"alert_level\"] = \"concern\"\n        elif \"7d\" in sr_data and sr_data[\"7d\"][\"sharpe\"] < self.baseline_sharpe * 0.5:\n            result[\"alert_level\"] = \"warning\"\n        \n        result[\"alerts\"] = alerts\n        \n        # Trend analysis on 7d Sharpe (if enough history)\n        if n > 2016 * 5:  # need 5+ rolling windows\n            result[\"trend\"] = self._compute_trend()\n        \n        return result\n    \n    def _compute_trend(self) -> dict:\n        \"\"\"Fit linear trend to recent rolling Sharpe values.\"\"\"\n        w = self.windows[0]  # use shortest window\n        returns = np.array(self._returns)\n        \n        # Compute rolling Sharpe at regular intervals (every 288 bars = daily)\n        sharpes = []\n        for end in range(w, len(returns), 288):\n            window = returns[end - w:end]\n            mean_r = np.mean(window)\n            std_r = np.std(window, ddof=1)\n            if std_r > 1e-12:\n                sr = (mean_r / std_r) * np.sqrt(self.annualization_factor)\n                sharpes.append(sr)\n        \n        if len(sharpes) < 5:\n            return {\"slope\": 0, \"p_value\": 1.0, \"decaying\": False}\n        \n        # Recent 30 data points (last 30 days of daily Sharpe)\n        recent = np.array(sharpes[-30:])\n        x = np.arange(len(recent))\n        \n        # Linear regression\n        n = len(recent)\n        x_mean = np.mean(x)\n        y_mean = np.mean(recent)\n        \n        ss_xy = np.sum((x - x_mean) * (recent - y_mean))\n        ss_xx = np.sum((x - x_mean) ** 2)\n        \n        if ss_xx < 1e-12:\n            return {\"slope\": 0, \"p_value\": 1.0, \"decaying\": False}\n        \n        slope = ss_xy / ss_xx\n        intercept = y_mean - slope * x_mean\n        \n        # Standard error and p-value for slope\n        residuals = recent - (intercept + slope * x)\n        mse = np.sum(residuals**2) / (n - 2)\n        se_slope = np.sqrt(mse / ss_xx)\n        \n        if se_slope < 1e-12:\n            t_stat = 0\n        else:\n            t_stat = slope / se_slope\n        \n        # Approximate p-value (two-sided)\n        from scipy import stats as scipy_stats\n        p_value = 2 * (1 - scipy_stats.t.cdf(abs(t_stat), n - 2))\n        \n        return {\n            \"slope\": slope,\n            \"slope_per_day\": slope,  # since we sampled daily\n            \"t_statistic\": t_stat,\n            \"p_value\": p_value,\n            \"decaying\": slope < 0 and p_value < 0.05,\n            \"estimated_days_to_zero\": -intercept / slope if slope < 0 and slope != 0 else float(\"inf\"),\n        }\n```\n\n### Recommended Thresholds\n\n| Metric | Warning | Concern | Critical |\n|--------|---------|---------|----------|\n| 7d Sharpe / Baseline Sharpe | < 0.50 | < 0.25 | < 0 |\n| 14d Sharpe / Baseline Sharpe | < 0.60 | < 0.40 | < 0.10 |\n| 30d Sharpe / Baseline Sharpe | < 0.70 | < 0.50 | < 0 |\n| Trend slope (p < 0.05) | < -0.05/day | < -0.10/day | < -0.20/day |\n\n---\n\n## 5. Sequential Probability Ratio Test (SPRT)\n\n### Mathematical Formulation\n\nWald's SPRT tests between two hypotheses sequentially:\n\n- $H_0$: win rate $p = p_0$ (strategy is working, e.g., $p_0 = 0.53$)\n- $H_1$: win rate $p = p_1$ (strategy has decayed, e.g., $p_1 = 0.50$)\n\nAfter each trade outcome $x_t \\in \\{0, 1\\}$ (loss or win):\n\n$$\\Lambda_t = \\Lambda_{t-1} + \\log \\frac{P(x_t | H_1)}{P(x_t | H_0)}$$\n\nFor Bernoulli outcomes:\n- If $x_t = 1$ (win): $\\Lambda_t += \\log(p_1 / p_0)$\n- If $x_t = 0$ (loss): $\\Lambda_t += \\log((1-p_1) / (1-p_0))$\n\nDecision boundaries:\n- Accept $H_1$ (edge gone) when: $\\Lambda_t \\ge \\log \\frac{1-\\beta}{\\alpha}$ (upper boundary $B$)\n- Accept $H_0$ (edge intact) when: $\\Lambda_t \\le \\log \\frac{\\beta}{1-\\alpha}$ (lower boundary $A$)\n\nWhere $\\alpha$ = probability of false alarm, $\\beta$ = probability of missing real decay.\n\n**Expected sample size:** SPRT is optimal — it requires the fewest observations on average to reach a decision with given error rates. Under $H_0$:\n\n$$E[N | H_0] \\approx \\frac{(1-\\alpha) \\log \\frac{1-\\alpha}{\\beta} + \\alpha \\log \\frac{\\alpha}{1-\\beta}}{p_0 \\log \\frac{p_0}{p_1} + (1-p_0) \\log \\frac{1-p_0}{1-p_1}}$$\n\nFor $p_0 = 0.53$, $p_1 = 0.50$, $\\alpha = \\beta = 0.05$: roughly 800-1200 trades to decide.\n\n**Truncated SPRT:** In practice, you don't want to wait forever. Truncate at $N_{max}$ observations and make a decision based on the sign of $\\Lambda_t$.\n\n### Python Implementation\n\n```python\n@dataclass \nclass SPRTDetector:\n    \"\"\"\n    Sequential Probability Ratio Test for win rate monitoring.\n    \n    Tests H0: p = p0 (edge intact) vs H1: p = p1 (edge decayed).\n    Optimal in terms of expected sample size for given error rates.\n    \"\"\"\n    p0: float = 0.53        # win rate under H0 (edge intact)\n    p1: float = 0.50        # win rate under H1 (edge gone)\n    alpha: float = 0.05     # false alarm probability\n    beta: float = 0.05      # missed detection probability\n    max_samples: int = 2000 # truncation point\n    \n    # Precomputed\n    _log_ratio_win: float = 0.0\n    _log_ratio_loss: float = 0.0\n    _upper_bound: float = 0.0\n    _lower_bound: float = 0.0\n    \n    # State\n    _lambda: float = 0.0\n    _n: int = 0\n    _wins: int = 0\n    _decisions: list = field(default_factory=list)\n    \n    def __post_init__(self):\n        self._log_ratio_win = math.log(self.p1 / self.p0)\n        self._log_ratio_loss = math.log((1 - self.p1) / (1 - self.p0))\n        self._upper_bound = math.log((1 - self.beta) / self.alpha)\n        self._lower_bound = math.log(self.beta / (1 - self.alpha))\n    \n    def update(self, is_win: bool) -> dict:\n        \"\"\"\n        Process one trade outcome.\n        \n        Args:\n            is_win: True if the trade was profitable\n        \"\"\"\n        self._n += 1\n        if is_win:\n            self._wins += 1\n            self._lambda += self._log_ratio_win\n        else:\n            self._lambda += self._log_ratio_loss\n        \n        decision = \"continue\"\n        \n        if self._lambda >= self._upper_bound:\n            decision = \"decay_detected\"  # Accept H1\n        elif self._lambda <= self._lower_bound:\n            decision = \"edge_intact\"     # Accept H0\n        elif self._n >= self.max_samples:\n            decision = \"decay_detected\" if self._lambda > 0 else \"edge_intact\"\n        \n        result = {\n            \"decision\": decision,\n            \"log_likelihood_ratio\": self._lambda,\n            \"upper_bound\": self._upper_bound,\n            \"lower_bound\": self._lower_bound,\n            \"n_samples\": self._n,\n            \"observed_win_rate\": self._wins / self._n,\n            \"progress_to_decay\": self._lambda / self._upper_bound,\n            \"progress_to_intact\": self._lambda / self._lower_bound,\n        }\n        \n        if decision != \"continue\":\n            self._decisions.append({\n                \"decision\": decision,\n                \"n_samples\": self._n,\n                \"win_rate\": self._wins / self._n,\n            })\n            # Reset for next test\n            self._lambda = 0.0\n            self._n = 0\n            self._wins = 0\n        \n        return result\n    \n    def expected_samples(self) -> dict:\n        \"\"\"Expected number of samples to reach a decision.\"\"\"\n        kl_h0 = (self.p0 * math.log(self.p0 / self.p1) + \n                  (1 - self.p0) * math.log((1 - self.p0) / (1 - self.p1)))\n        kl_h1 = (self.p1 * math.log(self.p1 / self.p0) + \n                  (1 - self.p1) * math.log((1 - self.p1) / (1 - self.p0)))\n        \n        e_n_h0 = ((1 - self.alpha) * self._lower_bound + \n                   self.alpha * self._upper_bound) / (-kl_h0) if kl_h0 != 0 else float(\"inf\")\n        e_n_h1 = ((1 - self.beta) * self._upper_bound + \n                   self.beta * self._lower_bound) / kl_h1 if kl_h1 != 0 else float(\"inf\")\n        \n        return {\n            \"under_h0_edge_intact\": abs(e_n_h0),\n            \"under_h1_edge_gone\": abs(e_n_h1),\n        }\n\n\n@dataclass\nclass ContinuousSPRT:\n    \"\"\"\n    SPRT for continuous returns (not just win/loss).\n    \n    Tests H0: mean return = mu0 vs H1: mean return = mu1.\n    Assumes returns are approximately normally distributed per period.\n    \"\"\"\n    mu0: float              # mean under H0 (edge intact)\n    mu1: float              # mean under H1 (edge gone, e.g., 0)\n    sigma: float            # known/estimated return volatility\n    alpha: float = 0.05\n    beta: float = 0.05\n    max_samples: int = 2000\n    \n    _lambda: float = 0.0\n    _n: int = 0\n    _upper_bound: float = 0.0\n    _lower_bound: float = 0.0\n    \n    def __post_init__(self):\n        self._upper_bound = math.log((1 - self.beta) / self.alpha)\n        self._lower_bound = math.log(self.beta / (1 - self.alpha))\n    \n    def update(self, return_value: float) -> dict:\n        self._n += 1\n        \n        # Log-likelihood ratio for normal distributions with known variance\n        self._lambda += ((self.mu1 - self.mu0) / self.sigma**2) * (\n            return_value - (self.mu0 + self.mu1) / 2\n        )\n        \n        decision = \"continue\"\n        if self._lambda >= self._upper_bound:\n            decision = \"decay_detected\"\n        elif self._lambda <= self._lower_bound:\n            decision = \"edge_intact\"\n        elif self._n >= self.max_samples:\n            decision = \"decay_detected\" if self._lambda > 0 else \"edge_intact\"\n        \n        result = {\n            \"decision\": decision,\n            \"log_likelihood_ratio\": self._lambda,\n            \"n_samples\": self._n,\n        }\n        \n        if decision != \"continue\":\n            self._lambda = 0.0\n            self._n = 0\n        \n        return result\n```\n\n### Calibration for 5-Min BTC\n\n- **p0 (H0 win rate)**: Use your actual observed win rate from validation period. For a model predicting direction, this might be 0.52-0.55.\n- **p1 (H1 win rate)**: 0.50 (random chance) or slightly below (0.49) if you want to detect complete edge loss.\n- **alpha**: 0.05 (5% false alarm rate). In practice you might want 0.01 if false alarms are very costly (unnecessary retraining).\n- **beta**: 0.05 (5% miss rate). Matching alpha is conventional.\n- **max_samples**: 2000 trades. At 288 bars/day, this is ~7 days.\n\n---\n\n## 6. Feature Drift Detection (PSI)\n\n### Mathematical Formulation\n\nPopulation Stability Index (PSI) measures how much a feature distribution has shifted from its training distribution:\n\n$$PSI = \\sum_{i=1}^{B} (p_i^{actual} - p_i^{expected}) \\cdot \\ln \\frac{p_i^{actual}}{p_i^{expected}}$$\n\nWhere:\n- $B$ = number of bins\n- $p_i^{expected}$ = proportion of training data in bin $i$\n- $p_i^{actual}$ = proportion of recent data in bin $i$\n\nPSI is a symmetric form of KL divergence: $PSI = KL(actual \\| expected) + KL(expected \\| actual)$.\n\n**Standard thresholds (from credit risk, widely used):**\n- PSI < 0.1: no significant drift\n- 0.1 ≤ PSI < 0.2: moderate drift (warning)\n- PSI ≥ 0.2: significant drift (action required)\n\n**However, for 5-min crypto data, these thresholds are too loose.** Crypto features (volatility, volume, order book imbalance) drift faster than credit data. Recommended crypto-specific thresholds:\n\n- PSI < 0.05: stable\n- 0.05 ≤ PSI < 0.10: moderate drift (monitor)\n- 0.10 ≤ PSI < 0.20: significant drift (reduce position size, consider retraining)\n- PSI ≥ 0.20: severe drift (halt predictions, retrain immediately)\n\n**Binning strategy matters:** Use quantile-based bins from the training distribution (deciles = 10 bins). This ensures equal density per bin in the reference distribution, making PSI more sensitive to shifts in the tails.\n\n**Multi-feature aggregation:** When monitoring many features, use:\n1. Maximum PSI across all features (most conservative)\n2. Mean PSI (overall drift measure)\n3. Count of features with PSI > threshold (breadth of drift)\n\n### Python Implementation\n\n```python\n@dataclass\nclass PSIMonitor:\n    \"\"\"\n    Population Stability Index monitor for feature drift detection.\n    \n    Computes PSI for each feature by comparing recent data against\n    the training distribution.\n    \"\"\"\n    n_bins: int = 10\n    window_size: int = 2016  # 7 days of 5-min bars\n    \n    # Learned from training data\n    _bin_edges: dict = field(default_factory=dict)       # feature_name -> bin edges\n    _reference_props: dict = field(default_factory=dict)  # feature_name -> proportions\n    \n    # Buffers for recent data\n    _buffers: dict = field(default_factory=dict)  # feature_name -> deque\n    \n    def fit(self, training_data: dict):\n        \"\"\"\n        Learn reference distributions from training data.\n        \n        Args:\n            training_data: dict of feature_name -> np.array of values\n        \"\"\"\n        for name, values in training_data.items():\n            values = np.array(values)\n            # Quantile-based bins\n            quantiles = np.linspace(0, 100, self.n_bins + 1)\n            edges = np.percentile(values, quantiles)\n            # Ensure unique edges\n            edges = np.unique(edges)\n            \n            counts, _ = np.histogram(values, bins=edges)\n            props = counts / counts.sum()\n            # Avoid zero proportions (would make log undefined)\n            props = np.clip(props, 1e-6, None)\n            props = props / props.sum()\n            \n            self._bin_edges[name] = edges\n            self._reference_props[name] = props\n            self._buffers[name] = deque(maxlen=self.window_size)\n    \n    def update(self, features: dict) -> dict:\n        \"\"\"\n        Process one observation of features.\n        \n        Args:\n            features: dict of feature_name -> value\n        \n        Returns:\n            PSI values and alerts for each feature.\n        \"\"\"\n        for name, value in features.items():\n            if name in self._buffers:\n                self._buffers[name].append(value)\n        \n        result = {\n            \"features\": {},\n            \"max_psi\": 0.0,\n            \"mean_psi\": 0.0,\n            \"drifted_features\": [],\n            \"alert_level\": \"stable\",\n        }\n        \n        psi_values = []\n        \n        for name in self._bin_edges:\n            buf = self._buffers.get(name)\n            if buf is None or len(buf) < self.window_size // 2:\n                continue\n            \n            recent = np.array(buf)\n            edges = self._bin_edges[name]\n            ref_props = self._reference_props[name]\n            \n            counts, _ = np.histogram(recent, bins=edges)\n            actual_props = counts / counts.sum()\n            actual_props = np.clip(actual_props, 1e-6, None)\n            actual_props = actual_props / actual_props.sum()\n            \n            # Handle edge case where binning produced different number of bins\n            min_len = min(len(ref_props), len(actual_props))\n            ref_p = ref_props[:min_len]\n            act_p = actual_props[:min_len]\n            \n            psi = float(np.sum((act_p - ref_p) * np.log(act_p / ref_p)))\n            \n            level = \"stable\"\n            if psi >= 0.20:\n                level = \"severe\"\n            elif psi >= 0.10:\n                level = \"significant\"\n            elif psi >= 0.05:\n                level = \"moderate\"\n            \n            result[\"features\"][name] = {\n                \"psi\": psi,\n                \"level\": level,\n                \"top_shifted_bins\": self._top_shifted_bins(ref_p, act_p, edges),\n            }\n            \n            psi_values.append(psi)\n            \n            if psi >= 0.10:\n                result[\"drifted_features\"].append(name)\n        \n        if psi_values:\n            result[\"max_psi\"] = float(max(psi_values))\n            result[\"mean_psi\"] = float(np.mean(psi_values))\n        \n        # Overall alert level\n        if result[\"max_psi\"] >= 0.20:\n            result[\"alert_level\"] = \"severe\"\n        elif result[\"max_psi\"] >= 0.10 or len(result[\"drifted_features\"]) >= 3:\n            result[\"alert_level\"] = \"significant\"\n        elif result[\"max_psi\"] >= 0.05:\n            result[\"alert_level\"] = \"moderate\"\n        \n        return result\n    \n    def _top_shifted_bins(\n        self, ref_props: np.ndarray, actual_props: np.ndarray, edges: np.ndarray, top_k: int = 3\n    ) -> list:\n        \"\"\"Identify which bins contributed most to PSI.\"\"\"\n        contributions = (actual_props - ref_props) * np.log(actual_props / ref_props)\n        top_indices = np.argsort(contributions)[-top_k:][::-1]\n        \n        result = []\n        for idx in top_indices:\n            if idx < len(edges) - 1:\n                result.append({\n                    \"bin_range\": (float(edges[idx]), float(edges[idx + 1])),\n                    \"psi_contribution\": float(contributions[idx]),\n                    \"ref_proportion\": float(ref_props[idx]),\n                    \"actual_proportion\": float(actual_props[idx]),\n                })\n        return result\n```\n\n---\n\n## 7. Prediction Distribution Monitoring\n\n### Mathematical Formulation\n\nMonitor the distribution of model predictions (not outcomes) over time. If the model begins predicting one class overwhelmingly, it may have encountered a distributional shift it cannot handle.\n\n**KL Divergence monitoring:**\n\n$$D_{KL}(P_{current} \\| P_{reference}) = \\sum_{i} P_{current}(i) \\log \\frac{P_{current}(i)}{P_{reference}(i)}$$\n\nFor continuous prediction scores (e.g., probability of price increase), discretize into bins and compare histograms.\n\n**Why this matters for a BTC prediction system:**\n\nIf your model outputs P(up) for each 5-min bar, the training distribution might be roughly uniform over [0.3, 0.7] with some tails. If the model suddenly outputs values clustered near 0.5 (uncertain about everything) or near 0.9 (overconfident bullish), something has changed. This can detect problems *before* they show up in returns, because:\n\n1. Feature drift causes prediction drift immediately\n2. But returns only degrade after positions are taken\n3. So prediction drift is a leading indicator\n\n**Complementary metrics:**\n- Mean prediction shift (simplest)\n- Prediction entropy: $H = -\\sum p(x) \\log p(x)$. Decreasing entropy means the model is becoming more confident (possibly overfit to recent data).\n- Prediction autocorrelation: if consecutive predictions are highly correlated, the model may be stuck.\n\n### Python Implementation\n\n```python\n@dataclass\nclass PredictionDistributionMonitor:\n    \"\"\"\n    Monitors the distribution of model predictions over time.\n    \n    Detects when predictions become anomalous relative to the\n    training-period prediction distribution.\n    \"\"\"\n    n_bins: int = 20\n    window_size: int = 2016     # 7 days\n    kl_warning: float = 0.05\n    kl_critical: float = 0.15\n    \n    _reference_hist: Optional[np.ndarray] = None\n    _reference_edges: Optional[np.ndarray] = None\n    _buffer: deque = field(default_factory=lambda: deque(maxlen=2016))\n    _n: int = 0\n    \n    def fit(self, training_predictions: np.ndarray):\n        \"\"\"Learn reference prediction distribution.\"\"\"\n        edges = np.linspace(0, 1, self.n_bins + 1)\n        counts, _ = np.histogram(training_predictions, bins=edges)\n        props = counts / counts.sum()\n        props = np.clip(props, 1e-6, None)\n        self._reference_hist = props / props.sum()\n        self._reference_edges = edges\n        self._buffer = deque(maxlen=self.window_size)\n    \n    def update(self, prediction: float) -> dict:\n        self._n += 1\n        self._buffer.append(prediction)\n        \n        result = {\n            \"n_observations\": self._n,\n            \"current_mean_prediction\": float(np.mean(self._buffer)),\n            \"alert_level\": \"normal\",\n        }\n        \n        if len(self._buffer) < self.window_size // 4 or self._reference_hist is None:\n            return result\n        \n        current = np.array(self._buffer)\n        counts, _ = np.histogram(current, bins=self._reference_edges)\n        current_hist = counts / counts.sum()\n        current_hist = np.clip(current_hist, 1e-6, None)\n        current_hist = current_hist / current_hist.sum()\n        \n        # KL divergence\n        kl_div = float(np.sum(current_hist * np.log(current_hist / self._reference_hist)))\n        \n        # Symmetric KL (Jensen-Shannon-like)\n        m = 0.5 * (current_hist + self._reference_hist)\n        js_div = 0.5 * np.sum(current_hist * np.log(current_hist / m)) + \\\n                 0.5 * np.sum(self._reference_hist * np.log(self._reference_hist / m))\n        \n        # Prediction entropy\n        entropy = float(-np.sum(current_hist * np.log(current_hist)))\n        \n        # Directional bias\n        bullish_fraction = float(np.mean(current > 0.5))\n        \n        alert_level = \"normal\"\n        alerts = []\n        \n        if kl_div >= self.kl_critical:\n            alert_level = \"critical\"\n            alerts.append(f\"KL divergence {kl_div:.4f} exceeds critical threshold\")\n        elif kl_div >= self.kl_warning:\n            alert_level = \"warning\"\n            alerts.append(f\"KL divergence {kl_div:.4f} exceeds warning threshold\")\n        \n        if bullish_fraction > 0.75 or bullish_fraction < 0.25:\n            alerts.append(f\"Extreme directional bias: {bullish_fraction:.1%} bullish\")\n            if alert_level == \"normal\":\n                alert_level = \"warning\"\n        \n        result.update({\n            \"kl_divergence\": kl_div,\n            \"js_divergence\": float(js_div),\n            \"prediction_entropy\": entropy,\n            \"bullish_fraction\": bullish_fraction,\n            \"alert_level\": alert_level,\n            \"alerts\": alerts,\n        })\n        \n        return result\n```\n\n---\n\n## 8. Concept Drift Detection (DDM, EDDM, STEPD)\n\n### Mathematical Formulation\n\nConcept drift means the relationship $P(Y|X)$ changes over time. Unlike feature drift ($P(X)$ changes) or label drift ($P(Y)$ changes), concept drift means the model's learned mapping is fundamentally wrong even if inputs look similar.\n\n**DDM (Drift Detection Method) — Gama et al., 2004:**\n\nTrack the error rate $p_t$ and its standard deviation $s_t = \\sqrt{p_t(1-p_t)/t}$ over a sliding window. Maintain the minimum observed $p_{min} + s_{min}$.\n\n- Warning: $p_t + s_t \\ge p_{min} + 2 \\cdot s_{min}$\n- Alarm: $p_t + s_t \\ge p_{min} + 3 \\cdot s_{min}$\n\nEssentially, DDM detects when the error rate is statistically significantly above its historical minimum. Simple, fast, but assumes errors are i.i.d. Bernoulli — problematic for financial data where errors are correlated.\n\n**EDDM (Early Drift Detection Method) — Baena-Garcia et al., 2006:**\n\nInstead of tracking error rate directly, tracks the *distance between consecutive errors*. If errors are becoming more frequent, the average distance between them shrinks.\n\nLet $d_t$ = number of observations between $(t-1)$-th and $t$-th error, and $d'_t = d_t / s_d$ (standardized).\n\n- Warning: $d'_t + s_{d'_t} < (d'_{max} + s_{d'_{max}}) \\times 0.95$\n- Alarm: $d'_t + s_{d'_t} < (d'_{max} + s_{d'_{max}}) \\times 0.90$\n\nEDDM is better at detecting slow, gradual concept drift because it's sensitive to the *pattern* of errors, not just their rate.\n\n**STEPD (Statistical Test of Equal Proportions) — Nishida & Yamauchi, 2007:**\n\nCompares error rate in a recent window to error rate in the overall history using a chi-squared test for equal proportions.\n\n$$\\chi^2 = \\frac{(p_{recent} - p_{overall})^2}{p_{overall}(1-p_{overall})(1/n_{recent} + 1/n_{overall})}$$\n\n- Warning: p-value < 0.05\n- Alarm: p-value < 0.003\n\n**Which works best for financial data?**\n\n| Method | Gradual drift | Sudden shift | Correlated errors | Speed |\n|--------|:---:|:---:|:---:|:---:|\n| DDM | Poor | Good | Poor | Fast |\n| EDDM | Good | Moderate | Moderate | Fast |\n| STEPD | Moderate | Good | Good | Moderate |\n| ADWIN | Good | Good | Good | Moderate |\n\n**Recommendation: EDDM + ADWIN ensemble.** EDDM catches gradual concept drift well, ADWIN catches sudden shifts. For financial data specifically, STEPD's chi-squared test is more robust to the autocorrelated errors common in trading strategies. Use at least two methods and alarm when either fires.\n\n### Python Implementation\n\n```python\n@dataclass\nclass DDMDetector:\n    \"\"\"Drift Detection Method for classification error Regime-Aware Alpha Decay\n\n### Mathematical Formulation\n\nAlpha (excess return) may decay in one market regime while persisting in another. A model trained during a trending market may perform poorly in mean-reverting conditions but still work fine when trends resume.\n\n**Regime identification approaches:**\n\n1. **Hidden Markov Model (HMM):** Fit a 2-3 state HMM on market features (volatility, trend strength, volume). Each state represents a regime.\n\n2. **Volatility-based regimes:** Simple and robust. Classify by realized volatility percentile:\n   - Low vol: RV < 25th percentile of training period\n   - Normal vol: 25th-75th percentile\n   - High vol: > 75th percentile\n\n3. **Trend-based regimes:** Based on moving average slope or ADX:\n   - Trending up: MA slope > threshold\n   - Ranging: |MA slope| < threshold\n   - Trending down: MA slope < -threshold\n\n**Decomposing performance by regime:**\n\nFor each regime $k$, track separate performance metrics:\n\n$$\\alpha_k = \\frac{1}{n_k} \\sum_{t: regime_t = k} r_t$$\n\n$$SR_k = \\frac{\\bar{r}_k}{\\sigma_k} \\sqrt{A}$$\n\nThen detect decay *within each regime separately* using the methods from sections 1-5. This prevents false alarms when the strategy's overall performance drops simply because the market shifted to a regime where the strategy was never expected to work.\n\n**Regime transition matrix monitoring:** Track P(regime at t+1 | regime at t). If the transition dynamics change (e.g., the market used to alternate between regimes but now stays stuck in one), this affects how much time the strategy spends in favorable conditions.\n\n### Python Implementation\n\n```python\n@dataclass\nclass RegimeClassifier:\n    \"\"\"Simple volatility-based regime classifier.\"\"\"\n    vol_window: int = 288       # Edge Half-Life Estimation\n\n### Mathematical Formulation\n\nModel alpha decay as an exponential process:\n\n$$\\alpha(t) = \\alpha_0 \\cdot e^{-\\lambda t}$$\n\nWhere:\n- $\\alpha_0$ = initial alpha (at deployment or last retraining)\n- $\\lambda$ = decay rate\n- Half-life: $t_{1/2} = \\frac{\\ln 2}{\\lambda}$\n\n**Estimation procedure:**\n\n1. Compute rolling alpha (e.g., rolling 1-day Sharpe ratio or rolling mean excess return) at regular intervals since model deployment.\n2. Fit exponential decay via log-linear regression:\n\n$$\\ln(\\alpha(t)) = \\ln(\\alpha_0) - \\lambda t$$\n\nThis only works when $\\alpha(t) > 0$ for all measured points, so filter or cap early.\n\n3. For more robust estimation with negative values (which happen when noise exceeds signal), use nonlinear least squares:\n\n$$\\min_{\\alpha_0, \\lambda} \\sum_t (SR_t - \\alpha_0 e^{-\\lambda t})^2$$\n\n**Typical half-lives from practitioner data (crypto ML strategies):**\n\n| Strategy Type | Typical Half-Life | Source/Basis |\n|--------------|-------------------|-------------|\n| HFT market-making | 1-7 days | Extremely competitive, signals get arbed quickly |\n| 5-min momentum/ML | 2-6 weeks | Moderate competition, features are not proprietary |\n| Daily ML predictions | 1-3 months | Slower signal, less competition |\n| Cross-exchange arb | Hours to days | Quickly competed away |\n| Sentiment/NLP alpha | 1-4 weeks | Rapidly incorporated by other participants |\n\n**For your 5-min BTC system:** Expect a half-life of 2-6 weeks. This means you should plan to retrain every 1-3 weeks (before losing half the edge). The actual half-life depends critically on:\n\n1. **How unique your features are** — public features (OHLCV, basic indicators) have shorter half-lives\n2. **How many others trade similar signals** — more competition = faster decay\n3. **Market regime stability** — stable regimes let edges persist; volatile regime changes kill edges fast\n\n### Python Implementation\n\n```python\nfrom scipy.optimize import curve_fit\n\n\n@dataclass\nclass EdgeHalfLifeEstimator:\n    \"\"\"\n    Estimates the half-life of strategy alpha decay.\n    \n    Fits an exponential decay model to rolling performance metrics.\n    \"\"\"\n    measurement_interval: int = 288  # measure alpha daily (288 5-min bars)\n    min_measurements: int = 10       # minimum data points for fitting\n    \n    _alpha_measurements: list = field(default_factory=list)\n    _timestamps: list = field(default_factory=list)\n    _returns_buffer: list = field(default_factory=list)\n    _measurement_counter: int = 0\n    \n    def update(self, return_value: float) -> Optional[dict]:\n        \"\"\"\n        Add one return. False Alarm Management\n\n### The Core Trade-off\n\nEvery detector faces the fundamental trade-off:\n\n$$\\text{Total Cost} = C_{FA} \\times P(FA) \\times N_{alerts} + C_{MD} \\times P(MD) \\times N_{decay\\_events}$$\n\nWhere:\n- $C_{FA}$ = cost of a false alarm (unnecessary retraining, lost trading time, computational cost)\n- $P(FA)$ = false alarm probability\n- $C_{MD}$ = cost of missed detection (continued losses, drawdown)\n- $P(MD)$ = missed detection probability\n\n**Estimating costs for a 5-min crypto system:**\n\n- **False alarm cost**: ~$200-500 (cloud compute for retraining + 2 days of reduced position size + engineer time for review). If you're running retraining automatically, maybe $50-100 in compute alone.\n- **Missed decay cost**: Depends on position size and duration. If the strategy loses 0.1% per day when decayed and you run $100K, that's $100/day. Over 2 weeks of missed detection: $1,400.\n\nTypically $C_{MD} \\gg C_{FA}$, which means you should err on the side of more false alarms (be more sensitive).\n\n### Multi-Detector Confirmation\n\nThe most effective false alarm reduction strategy is requiring confirmation from independent detectors:\n\n**Independence principle:** CUSUM on returns and PSI on features are measuring different things. If both alarm, the probability of a simultaneous false alarm is $P(FA_1) \\times P(FA_2)$, which is much smaller than either alone (assuming independence, which is approximately true for these two).\n\n**Confirmation window:** Don't require all detectors to fire simultaneously. Use a window:\n\n```\nIf 2+ detectors alarm within a 24-hour window (288 bars):\n    -> Treat as confirmed signal\nElse:\n    -> Single detector alarm, increase monitoring only\n```\n\n### Implementation\n\n```python\n@dataclass\nclass FalseAlarmManager:\n    \"\"\"\n    Manages multiple detector outputs to minimize false alarms\n    while maintaining sensitivity to real decay.\n    \n    Uses a confirmation window approach: requires multiple\n    independent detectors to agree within a time window.\n    \"\"\"\n    confirmation_window: int = 288  # 24 hours\n    min_confirming_detectors: int = 2\n    cooldown_period: int = 576      # 48 hours after an alarm before re-alarming\n    \n    # Cost parameters for threshold optimization\n    cost_false_alarm: float = 200.0\n    cost_missed_decay_per_bar: float = 0.5\n    \n    # State\n    _recent_alarms: dict = field(default_factory=dict)  # detector -> list of timestamps\n    _confirmed_alarms: list = field(default_factory=list)\n    _last_confirmed_at: int = 0\n    _n: int = 0\n    \n    def process_detector_output(self, detector_name: str, result: dict) -> dict:\n        \"\"\"\n        Process output from one detector. Estimate detection delay on decayed data\n            for _ in range(n_simulations):\n                detector = detector_class(**params)\n                \n                for t, r in enumerate(returns_decayed[:detection_horizon]):\n                    result = detector.update(r)\n                    if result.get(\"alarm\") or result.get(\"change_detected\"):\n                        detection_delays.append(t)\n                        break\n                else:\n                    detection_delays.append(detection_horizon)\n            \n            avg_delay = np.mean(detection_delays)\n            miss_rate = sum(1 for d in detection_delays if d >= detection_horizon) / n_simulations\n            \n            #
