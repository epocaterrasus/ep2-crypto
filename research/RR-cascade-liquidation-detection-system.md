Historical Cascade Labeling\n\n### The Labeling Challenge\n\nThere is no canonical definition of a \"liquidation cascade.\" We need to create labels that are:\n- **Consistent**: Same criteria applied everywhere\n- **Meaningful**: Capture events that matter for trading\n- **Sensitive enough**: Don't miss real cascades\n- **Specific enough**: Don't label every volatile period as a cascade\n\n### Multi-Criteria Labeling\n\n```python\nimport numpy as np\nimport pandas as pd\nfrom dataclasses import dataclass\nimport logging\n\nlogger = logging.getLogger(__name__)\n\n\n@dataclass\nclass CascadeLabelConfig:\n    \"\"\"\n    Thresholds for labeling a 5-minute period as a cascade.\n    ALL conditions must be met simultaneously.\n    \"\"\"\n    min_price_move_pct: float = 1.5       # Price move: absolute percentage change\n        result[\"price_move_pct\"] = (\n            (result[\"close\"] - result[\"open\"]).abs() / result[\"open\"] * 100\n        )\n\n        # Also consider high-low range for wicking cascades\n        result[\"range_pct\"] = (\n            (result[\"high\"] - result[\"low\"]) / result[\"open\"] * 100\n        )\n        # need sufficient history\n            return 1.0  # neutral\n\n        if self.latest_snapshot is None:\n            return 1.0\n\n        current_depth = self.latest_snapshot.total_depth_2pct\n        avg_depth = np.mean([s.total_depth_2pct for s in self.history])\n\n        if avg_depth == 0:\n            return 1.0\n\n        return current_depth / avg_depth\n\n    def is_thin(self, threshold: float = 0.5) -> bool:\n        \"\"\"True if current depth is below threshold × average depth.\"\"\"\n        return self.compute_depth_ratio() < threshold\n\n    def get_directional_vulnerability(self) -> dict[str, float]:\n        \"\"\"\n        Which side of the book is thinner?\n        If bid side is thin, downward cascades are more dangerous.\n        If ask side is thin, upward short squeezes are more dangerous.\n        \"\"\"\n        if self.latest_snapshot is None or len(self.history) < 100:\n            return {\"bid_ratio\": 1.0, \"ask_ratio\": 1.0, \"vulnerable_side\": \"none\"}\n\n        avg_bid = np.mean([s.bid_depth_2pct_usd for s in self.history])\n        avg_ask = np.mean([s.ask_depth_2pct_usd for s in self.history])\n\n        bid_ratio = self.latest_snapshot.bid_depth_2pct_usd / avg_bid if avg_bid > 0 else 1.0\n        ask_ratio = self.latest_snapshot.ask_depth_2pct_usd / avg_ask if avg_ask > 0 else 1.0\n\n        if bid_ratio < ask_ratio:\n            vulnerable = \"downside\"  # thin bids = vulnerable to downward cascade\n        else:\n            vulnerable = \"upside\"    # thin asks = vulnerable to upward squeeze\n\n        return {\n            \"bid_ratio\": bid_ratio,\n            \"ask_ratio\": ask_ratio,\n            \"vulnerable_side\": vulnerable,\n        }\n\n    async def connect_ws(self) -> None:\n        \"\"\"Connect to Binance depth stream for real-time updates.\"\"\"\n        url = self.BINANCE_WS_URL.format(symbol=self.symbol)\n        self._running = True\n\n        while self._running:\n            try:\n                async with websockets.connect(url, ping_interval=20) as ws:\n                    logger.info(\"Connected to depth stream for %s\", self.symbol)\n\n                    last_process_time = 0.0\n                    async for raw_msg in ws:\n                        now = time.time()\n                        # Throttle processing to snapshot_interval_sec\n                        if now - last_process_time < self.snapshot_interval_sec:\n                            continue\n\n                        try:\n                            msg = json.loads(raw_msg)\n                            bids = msg.get(\"b\", msg.get(\"bids\", []))\n                            asks = msg.get(\"a\", msg.get(\"asks\", []))\n                            snapshot = self.process_depth_message(bids, asks)\n                            if snapshot:\n                                last_process_time = now\n                        except (json.JSONDecodeError, KeyError):\n                            logger.warning(\"Failed to parse depth message\")\n\n            except websockets.ConnectionClosed:\n                logger.warning(\"Depth WS closed, reconnecting in 3s\")\n                await asyncio.sleep(3)\n            except Exception:\n                logger.exception(\"Depth WS error, reconnecting in 5s\")\n                await asyncio.sleep(5)\n\n    def stop(self) -> None:\n        self._running = False\n```\n\n---\n\n## 10. self._halt_time\n            if elapsed < self._min_halt_duration_sec:\n                return True, f\"HALTED (resume in {self._min_halt_duration_sec - elapsed:.0f}s): {self._halt_reason}\"\n            else:\n                self._halted = False\n                self._halt_reason = \"\"\n                logger.info(\"Circuit breaker released\")\n\n        return False, \"\"\n```\n\n---\n\n## 11. False Positive Management\n\n### The False Positive Problem\n\nIn practice, \"pre-cascade\" conditions are common but actual cascades are rare. A system that alerts on every elevated signal will generate 10–20 false alerts per real cascade, leading to alert fatigue and over-hedging.\n\n### Multi-Layer Filtering Strategy\n\n```python\nimport numpy as np\nfrom collections import deque\nfrom dataclasses import dataclass\nimport time\nimport logging\n\nlogger = logging.getLogger(__name__)\n\n\n@dataclass\nclass AlertRecord:\n    timestamp: float\n    cascade_prob: float\n    hawkes_ratio: float\n    actual_cascade: bool | None = None  # filled in retrospectively\n    alert_level: str = \"info\"\n\n\nclass FalsePositiveManager:\n    \"\"\"\n    Multi-layer approach to minimize false positives:\n\n    1. **Confirmation window**: Don't alert on first spike. Wait for\n       sustained elevation (>30 seconds of cascade_prob > threshold).\n\n    2. **Multi-signal confirmation**: Require 3+ independent signals\n       (Hawkes, OI, funding, depth, burst rate) to be elevated simultaneously.\n\n    3. **Cooldown**: After an alert, suppress new alerts for 10 minutes\n       unless conditions dramatically worsen.\n\n    4. **Adaptive threshold**: Track precision over rolling 30-day window.\n       If precision drops below target, raise the threshold.\n\n    5. **Time-of-day weighting**: Cascades cluster around funding times\n       (00:00, 08:00, 16:00 UTC) and US market hours. Weight signals\n       higher during these periods.\n    \"\"\"\n\n    def __init__(\n        self,\n        base_threshold: float = 0.6,\n        confirmation_window_sec: float = 30.0,\n        cooldown_sec: float = 600.0,\n        target_precision: float = 0.5,\n    ):\n        self.base_threshold = base_threshold\n        self.confirmation_window_sec = confirmation_window_sec\n        self.cooldown_sec = cooldown_sec\n        self.target_precision = target_precision\n\n        # Track recent probability readings for confirmation\n        self.prob_history: deque[tuple[float, float]] = deque(maxlen=1000)\n        # (timestamp, probability)\n\n        # Alert history for precision tracking\n        self.alert_history: deque[AlertRecord] = deque(maxlen=500)\n        self.last_alert_time: float = 0.0\n\n        # Adaptive threshold state\n        self.current_threshold = base_threshold\n\n    def update(\n        self,\n        cascade_prob: float,\n        hawkes_ratio: float,\n        depth_ratio: float,\n        burst_rate: float,\n        funding_zscore: float,\n        oi_percentile: float,\n    ) -> tuple[bool, str, float]:\n        \"\"\"\n        Process new cascade probability reading.\n\n        Returns:\n            (should_alert, alert_level, adjusted_probability)\n            alert_level: \"info\", \"warning\", \"critical\"\n        \"\"\"\n        now = time.time()\n        self.prob_history.append((now, cascade_prob))\n\n        # Layer 1: Below threshold — no alert\n        if cascade_prob < self.current_threshold:\n            return False, \"info\", cascade_prob\n\n        # Layer 2: Cooldown check\n        if now - self.last_alert_time < self.cooldown_sec:\n            # Allow override for dramatically worse conditions\n            if cascade_prob < 0.9:\n                return False, \"info\", cascade_prob\n\n        # Layer 3: Confirmation window\n        # Check that prob has been elevated for at least confirmation_window_sec\n        sustained = self._check_sustained_elevation(now)\n        if not sustained:\n            return False, \"info\", cascade_prob\n\n        # Layer 4: Multi-signal confirmation\n        n_elevated = self._count_elevated_signals(\n            hawkes_ratio, depth_ratio, burst_rate, funding_zscore, oi_percentile\n        )\n        if n_elevated < 3:\n            return False, \"info\", cascade_prob\n\n        # Layer 5: Time-of-day weight\n        time_weight = self._time_of_day_weight(now)\n        adjusted_prob = cascade_prob * time_weight\n\n        if adjusted_prob < self.current_threshold:\n            return False, \"info\", adjusted_prob\n\n        # Alert!\n        alert_level = \"critical\" if adjusted_prob > 0.8 else \"warning\"\n        self.last_alert_time = now\n        self.alert_history.append(AlertRecord(\n            timestamp=now,\n            cascade_prob=adjusted_prob,\n            hawkes_ratio=hawkes_ratio,\n            alert_level=alert_level,\n        ))\n\n        logger.warning(\n            \"CASCADE ALERT [%s]: prob=%.3f (adj=%.3f), hawkes=%.1fx, \"\n            \"depth=%.2f, burst=%.1f/s, funding_z=%.1f, oi_pct=%.0f, \"\n            \"signals=%d/5\",\n            alert_level, cascade_prob, adjusted_prob, hawkes_ratio,\n            depth_ratio, burst_rate, funding_zscore, oi_percentile,\n            n_elevated,\n        )\n\n        return True, alert_level, adjusted_prob\n\n    def _check_sustained_elevation(self, now: float) -> bool:\n        \"\"\"Check that probability has been above threshold for the full window.\"\"\"\n        cutoff = now - self.confirmation_window_sec\n        recent = [p for t, p in self.prob_history if t >= cutoff]\n\n        if len(recent) < 3:  # need at least 3 readings\n            return False\n\n        # At least 80% of readings must be above threshold\n        above = sum(1 for p in recent if p >= self.current_threshold)\n        return above / len(recent) >= 0.8\n\n    def _count_elevated_signals(\n        self,\n        hawkes_ratio: float,\n        depth_ratio: float,\n        burst_rate: float,\n        funding_zscore: float,\n        oi_percentile: float,\n    ) -> int:\n        \"\"\"Count how many independent signals are in elevated state.\"\"\"\n        count = 0\n        if hawkes_ratio > 3.0:\n            count += 1\n        if depth_ratio < 0.5:\n            count += 1\n        if burst_rate > 5.0:\n            count += 1\n        if abs(funding_zscore) > 1.5:\n            count += 1\n        if oi_percentile > 80:\n            count += 1\n        return count\n\n    def _time_of_day_weight(self, now: float) -> float:\n        \"\"\"\n        Weight based on time of day. Cascades are more likely:\n        - Around funding times (00:00, 08:00, 16:00 UTC) ±1 hour\n        - During US market hours (14:00-21:00 UTC)\n        - During Asian open (01:00-03:00 UTC)\n\n        Returns weight 0.8-1.2 (never suppresses by more than 20%).\n        \"\"\"\n        from datetime import datetime, timezone\n\n        utc_hour = datetime.fromtimestamp(now, tz=timezone.utc).hour\n\n        # High-risk hours\n        high_risk = {0, 1, 7, 8, 9, 15, 16, 17, 14, 18, 19, 20, 21}\n        if utc_hour in high_risk:\n            return 1.1\n\n        # Low-risk hours (typically quieter)\n        low_risk = {3, 4, 5, 6, 10, 11, 12, 13}\n        if utc_hour in low_risk:\n            return 0.9\n\n        return 1.0\n\n    def label_retrospective(self, alert_time: float, was_cascade: bool) -> None:\n        \"\"\"\n        After the fact, label whether an alert was a true or false positive.\n        Used to track precision and adapt threshold.\n        \"\"\"\n        for record in self.alert_history:\n            if abs(record.timestamp - alert_time) < 60:\n                record.actual_cascade = was_cascade\n                break\n\n        self._adapt_threshold()\n\n    def _adapt_threshold(self) -> None:\n        \"\"\"\n        Adjust threshold based on rolling precision.\n        If precision < target, raise threshold. If precision > target + 0.1, lower it.\n        \"\"\"\n        labeled = [r for r in self.alert_history if r.actual_cascade is not None]\n        if len(labeled) < 20:  # need sufficient samples\n            return\n\n        recent = labeled[-50:]  # last 50 labeled alerts\n        true_positives = sum(1 for r in recent if r.actual_cascade)\n        precision = true_positives / len(recent)\n\n        if precision < self.target_precision:\n            self.current_threshold = min(self.current_threshold + 0.02, 0.95)\n            logger.info(\n                \"Raising threshold to %.3f (precision=%.2f < target=%.2f)\",\n                self.current_threshold, precision, self.target_precision,\n            )\n        elif precision > self.target_precision + 0.15:\n            self.current_threshold = max(self.current_threshold - 0.01, self.base_threshold)\n            logger.info(\n                \"Lowering threshold to %.3f (precision=%.2f well above target)\",\n                self.current_threshold, precision,\n            )\n\n    def get_stats(self) -> dict:\n        labeled = [r for r in self.alert_history if r.actual_cascade is not None]\n        tp = sum(1 for r in labeled if r.actual_cascade)\n        fp = len(labeled) - tp\n\n        return {\n            \"total_alerts\": len(self.alert_history),\n            \"labeled_alerts\": len(labeled),\n            \"true_positives\": tp,\n            \"false_positives\": fp,\n            \"precision\": tp / len(labeled) if labeled else 0.0,\n            \"current_threshold\": self.current_threshold,\n        }\n```\n\n### Expected False Positive Rates\n\n| Strategy | FP per real cascade | Detection delay |\n|----------|-------------------|----------------|\n| Naive threshold (prob > 0.5) | 15–25 FP/TP | Immediate |\n| + Confirmation window (30s) | 8–12 FP/TP | +30s |\n| + Multi-signal (3/5) | 3–5 FP/TP | +30s |\n| + Adaptive threshold | 2–4 FP/TP | +30–60s |\n| + Time weighting | 1.5–3 FP/TP | +30–60s |\n\nThe tradeoff is clear: each layer reduces false positives at the cost of detection latency. For a trading system, the confirmation window + multi-signal approach (3–5 FP/TP, 30s delay) is typically the best balance.\n\n---\n\n## 12. Cascade Direction Prediction\n\n### Concept\n\nOnce a cascade is flagged as likely, predicting its direction (long liquidation = price down, short liquidation = price up) is critical for positioning. Direction depends on which side has more \"fuel\" (positions to liquidate).\n\n### Key Predictors\n\n1. **Funding rate sign**: Positive funding → longs paying shorts → more longs → long liquidation cascade likely\n2. **OI-weighted funding**: High OI × positive funding = massive long exposure\n3. **Liquidation imbalance in last 60s**: If early liquidations are already skewing long, cascade continues\n4. **Heatmap asymmetry**: More liquidation volume below current price (long liquidation levels) vs above\n5. **CVD (Cumulative Volume Delta)**: Aggressive buying in recent hours → long buildup → vulnerable to long liquidation\n\n### Implementation\n\n```python\nimport numpy as np\nfrom dataclasses import dataclass\nimport logging\n\nlogger = logging.getLogger(__name__)\n\n\n@dataclass\nclass DirectionPrediction:\n    direction: str          # \"long_liquidation\" (price down) or \"short_liquidation\" (price up)\n    confidence: float       # 0.5 = coin flip, 1.0 = certain\n    contributing_factors: dict[str, float]\n\n\nclass CascadeDirectionPredictor:\n    \"\"\"\n    Predicts the direction of an impending cascade using a weighted\n    evidence combination approach.\n\n    Each signal contributes a directional vote with a weight based on\n    historical predictive power.\n    \"\"\"\n\n    def __init__(self):\n        # Weights learned from backtesting (can be recalibrated)\n        self.signal_weights = {\n            \"funding_rate\": 0.25,\n            \"liq_imbalance_60s\": 0.30,\n            \"heatmap_asymmetry\": 0.20,\n            \"oi_change_direction\": 0.15,\n            \"depth_asymmetry\": 0.10,\n        }\n\n    def predict(\n        self,\n        funding_rate: float,\n        funding_zscore: float,\n        liq_imbalance_60s: float,\n        heatmap_long_vol_nearby: float,\n        heatmap_short_vol_nearby: float,\n        oi_change_rate: float,\n        bid_depth_ratio: float,\n        ask_depth_ratio: float,\n    ) -> DirectionPrediction:\n        \"\"\"\n        Predict cascade direction.\n\n        Convention:\n        - Positive score → long liquidation expected (price will drop)\n        - Negative score → short liquidation expected (price will rise)\n\n        Each signal is normalized to [-1, +1] range.\n        \"\"\"\n        signals = {}\n\n        # 1. Funding rate signal\n        # Positive funding → longs pay shorts → long-heavy market → long liquidation likely\n        if abs(funding_zscore) > 0.5:\n            signals[\"funding_rate\"] = np.clip(funding_zscore / 3.0, -1.0, 1.0)\n        else:\n            signals[\"funding_rate\"] = 0.0\n\n        # 2. Liquidation imbalance (most important real-time signal)\n        # Positive imbalance = more long liquidations happening → cascade continues down\n        signals[\"liq_imbalance_60s\"] = np.clip(liq_imbalance_60s, -1.0, 1.0)\n\n        # 3. Heatmap asymmetry\n        # More long liquidation volume nearby → bigger long cascade potential\n        total_nearby = heatmap_long_vol_nearby + heatmap_short_vol_nearby\n        if total_nearby > 0:\n            signals[\"heatmap_asymmetry\"] = (\n                (heatmap_long_vol_nearby - heatmap_short_vol_nearby) / total_nearby\n            )\n        else:\n            signals[\"heatmap_asymmetry\"] = 0.0\n\n        # 4.
