Uncertainty weighting: L_total = (1/2σ²) * L_task + log(σ)\n        # This automatically balances tasks based on their noise level\n        w_dir = torch.exp(-self.log_var_direction)\n        w_vol = torch.exp(-self.log_var_volatility)\n\n        total = (\n            w_dir * L_dir + self.log_var_direction +\n            w_vol * L_vol + self.log_var_volatility\n        )\n\n        return {\n            \"total\": total,\n            \"direction\": L_dir,\n            \"volatility\": L_vol,\n            \"weight_direction\": w_dir.item(),\n            \"weight_volatility\": w_vol.item(),\n        }\n\n\n# --- Multi-task dataset ---\nclass MultiTaskCryptoDataset(Dataset):\n    \"\"\"Dataset that returns both direction labels and volatility targets.\"\"\"\n\n    def __init__(\n        self,\n        df: pd.DataFrame,\n        feature_cols: list[str],\n        direction_col: str,\n        seq_len: int = 48,\n        vol_window: int = 6,  # realized vol over next 6 bars (30 min)\n        return_col: str = \"log_return\",\n    ):\n        self.seq_len = seq_len\n        self.features = df[feature_cols].values.astype(np.float32)\n        self.directions = df[direction_col].values.astype(np.int64)\n\n        # Pre-compute forward realized volatility (std of next N returns)\n        returns = df[return_col].values.astype(np.float32)\n        self.volatility = np.array([\n            returns[i:i + vol_window].std() if i + vol_window <= len(returns) else 0.0\n            for i in range(len(returns))\n        ], dtype=np.float32)\n\n        max_start = len(df) - seq_len - vol_window\n        self.indices = list(range(0, max_start))\n\n    def __len__(self) -> int:\n        return len(self.indices)\n\n    def __getitem__(self, idx: int):\n        start = self.indices[idx]\n        end = start + self.seq_len\n\n        x = torch.from_numpy(self.features[start:end])\n        y_dir = torch.tensor(self.directions[end])\n        y_vol = torch.tensor(self.volatility[end])\n\n        return x, y_dir, y_vol\n```\n\n---\n\n## 8. Transfer Learning for Crypto\n\n### Does It Help?\n\nTransfer learning for crypto is an active research area with mixed results:\n\n| Transfer Strategy | Evidence | When It Helps |\n|---|---|---|\n| **1H -> 5M** (same asset) | Moderate positive | The 1H model learns coarser patterns (trend, support/resistance). Fine-tuning on 5M data adds microstructure. Works best when 5M training data is limited (<1 month). |\n| **ETH -> BTC** (cross-asset) | Weak positive | Crypto assets are highly correlated (0.7-0.9 rolling correlation). The shared dynamics (momentum, mean-reversion timing) transfer. But the model must unlearn asset-specific vol and liquidity patterns. |\n| **Simulated -> Real** | Not recommended | Simulated crypto data lacks the fat tails, volatility clustering, and microstructure effects of real data. |\n\n```python\nclass TransferLearningPipeline:\n    \"\"\"Pre-train on source domain, fine-tune on target domain.\n\n    Key decisions:\n    1. Which layers to freeze during fine-tuning\n    2. Learning rate for frozen vs unfrozen layers\n    3. How long to fine-tune (too long = lose pre-trained knowledge)\n    \"\"\"\n\n    def __init__(self, model: nn.Module, device: torch.device):\n        self.model = model\n        self.device = device\n\n    def freeze_backbone(self):\n        \"\"\"Freeze GRU and input projection. Only train the head.\"\"\"\n        for name, param in self.model.named_parameters():\n            if \"head\" not in name:\n                param.requires_grad = False\n        trainable = sum(p.numel() for p in self.model.parameters() if p.requires_grad)\n        total = sum(p.numel() for p in self.model.parameters())\n        logger.info(\"Frozen backbone. Trainable: %d / %d params\", trainable, total)\n\n    def unfreeze_with_discriminative_lr(\n        self,\n        base_lr: float = 1e-4,\n        head_lr: float = 1e-3,\n    ) -> optim.AdamW:\n        \"\"\"Unfreeze all layers with different learning rates.\n\n        Lower LR for pre-trained layers (preserve learned features).\n        Higher LR for head (adapt to new task/asset).\n        \"\"\"\n        for param in self.model.parameters():\n            param.requires_grad = True\n\n        param_groups = [\n            {\n                \"params\": [p for n, p in self.model.named_parameters() if \"head\" not in n],\n                \"lr\": base_lr,\n                \"name\": \"backbone\",\n            },\n            {\n                \"params\": [p for n, p in self.model.named_parameters() if \"head\" in n],\n                \"lr\": head_lr,\n                \"name\": \"head\",\n            },\n        ]\n\n        return optim.AdamW(param_groups, weight_decay=1e-4)\n\n    def fine_tune(\n        self,\n        source_checkpoint: str,\n        target_train_loader: DataLoader,\n        target_val_loader: DataLoader,\n        phase1_epochs: int = 5,   # frozen backbone, train head only\n        phase2_epochs: int = 20,  # all layers with discriminative LR\n    ):\n        \"\"\"Two-phase fine-tuning strategy.\"\"\"\n        # Load pre-trained weights\n        checkpoint = torch.load(source_checkpoint, map_location=self.device)\n        self.model.load_state_dict(checkpoint[\"model_state_dict\"], strict=False)\n        logger.info(\"Loaded pre-trained weights from %s\", source_checkpoint)\n\n        self.model.to(self.device)\n        criterion = CostAwareLoss()\n\n        # Phase 1: Frozen backbone, train head to convergence\n        logger.info(\"Phase 1: Training head only (%d epochs)\", phase1_epochs)\n        self.freeze_backbone()\n        optimizer = optim.AdamW(\n            filter(lambda p: p.requires_grad, self.model.parameters()),\n            lr=1e-3, weight_decay=1e-4,\n        )\n\n        for epoch in range(phase1_epochs):\n            self.model.train()\n            total_loss = 0\n            for x, y in target_train_loader:\n                x, y = x.to(self.device), y.to(self.device)\n                out = self.model(x)\n                loss = criterion(out[\"logits\"], y)\n                optimizer.zero_grad(set_to_none=True)\n                loss.backward()\n                optimizer.step()\n                total_loss += loss.item()\n            logger.info(\"Phase 1 epoch %d: loss=%.4f\", epoch, total_loss / len(target_train_loader))\n\n        # Phase 2: Unfreeze all, discriminative LR\n        logger.info(\"Phase 2: Full fine-tuning (%d epochs)\", phase2_epochs)\n        optimizer = self.unfreeze_with_discriminative_lr(base_lr=1e-4, head_lr=5e-4)\n        scheduler = CosineAnnealingWarmRestarts(optimizer, T_0=5, T_mult=2, eta_min=1e-6)\n\n        best_val_loss = float(\"inf\")\n        for epoch in range(phase2_epochs):\n            self.model.train()\n            total_loss = 0\n            for x, y in target_train_loader:\n                x, y = x.to(self.device), y.to(self.device)\n                out = self.model(x)\n                loss = criterion(out[\"logits\"], y)\n                optimizer.zero_grad(set_to_none=True)\n                loss.backward()\n                torch.nn.utils.clip_grad_norm_(self.model.parameters(), max_norm=1.0)\n                optimizer.step()\n                total_loss += loss.item()\n\n            scheduler.step()\n\n            # Validate\n            self.model.eval()\n            val_loss = 0\n            with torch.no_grad():\n                for x, y in target_val_loader:\n                    x, y = x.to(self.device), y.to(self.device)\n                    out = self.model(x)\n                    val_loss += criterion(out[\"logits\"], y).item()\n            val_loss /= len(target_val_loader)\n\n            logger.info(\"Phase 2 epoch %d: train_loss=%.4f val_loss=%.4f\", epoch, total_loss / len(target_train_loader), val_loss)\n\n            if val_loss < best_val_loss:\n                best_val_loss = val_loss\n                torch.save(self.model.state_dict(), \"checkpoints/fine_tuned_best.pt\")\n```\n\n---\n\n## 9. Online/Incremental Training\n\n```python\nclass OnlineLearner:\n    \"\"\"Incremental training for GRU on new crypto data without catastrophic forgetting.\n\n    Three strategies implemented:\n    1. Replay buffer: store representative old samples, mix with new data\n    2. Elastic Weight Consolidation (EWC): penalize changes to important weights\n    3. Learning rate warmup: gentle adaptation to new distribution\n\n    For 5-min crypto, new data arrives constantly. You want to adapt to regime\n    changes without forgetting stable patterns.\n    \"\"\"\n\n    def __init__(\n        self,\n        model: nn.Module,\n        device: torch.device,\n        replay_buffer_size: int = 10000,\n        ewc_lambda: float = 100.0,\n        base_lr: float = 1e-4,\n        warmup_steps: int = 100,\n    ):\n        self.model = model\n        self.device = device\n        self.ewc_lambda = ewc_lambda\n        self.base_lr = base_lr\n        self.warmup_steps = warmup_steps\n\n        # Replay buffer (reservoir sampling)\n        self.replay_buffer_x: list[torch.Tensor] = []\n        self.replay_buffer_y: list[torch.Tensor] = []\n        self.replay_buffer_size = replay_buffer_size\n        self.total_seen = 0\n\n        # EWC: Fisher information and reference parameters\n        self.fisher_info: dict[str, torch.Tensor] = {}\n        self.reference_params: dict[str, torch.Tensor] = {}\n        self.ewc_initialized = False\n\n    def compute_fisher_information(self, dataloader: DataLoader, num_samples: int = 2000):\n        \"\"\"Compute diagonal Fisher information matrix.\n\n        Fisher information tells us which parameters are important for the\n        current task. We penalize changes to high-Fisher parameters when\n        learning new data.\n        \"\"\"\n        self.model.eval()\n        fisher = {n: torch.zeros_like(p) for n, p in self.model.named_parameters() if p.requires_grad}\n        reference = {n: p.data.clone() for n, p in self.model.named_parameters() if p.requires_grad}\n\n        criterion = nn.CrossEntropyLoss()\n        count = 0\n\n        for x, y in dataloader:\n            if count >= num_samples:\n                break\n            x, y = x.to(self.device), y.to(self.device)\n            self.model.zero_grad()\n            out = self.model(x)\n            loss = criterion(out[\"logits\"], y)\n            loss.backward()\n\n            for n, p in self.model.named_parameters():\n                if p.requires_grad and p.grad is not None:\n                    fisher[n] += (p.grad.data ** 2) * x.size(0)\n\n            count += x.size(0)\n\n        # Normalize\n        for n in fisher:\n            fisher[n] /= count\n\n        self.fisher_info = fisher\n        self.reference_params = reference\n        self.ewc_initialized = True\n        logger.info(\"Computed Fisher information over %d samples\", count)\n\n    def ewc_penalty(self) -> torch.Tensor:\n        \"\"\"EWC regularization: penalize deviation from reference parameters\n        weighted by Fisher information.\"\"\"\n        if not self.ewc_initialized:\n            return torch.tensor(0.0, device=self.device)\n\n        penalty = torch.tensor(0.0, device=self.device)\n        for n, p in self.model.named_parameters():\n            if n in self.fisher_info:\n                penalty += (self.fisher_info[n] * (p - self.reference_params[n]) ** 2).sum()\n\n        return penalty\n\n    def add_to_replay_buffer(self, x: torch.Tensor, y: torch.Tensor):\n        \"\"\"Reservoir sampling: maintains a uniform random sample of all seen data.\"\"\"\n        for i in range(x.size(0)):\n            self.total_seen += 1\n            if len(self.replay_buffer_x) < self.replay_buffer_size:\n                self.replay_buffer_x.append(x[i].cpu())\n                self.replay_buffer_y.append(y[i].cpu())\n            else:\n                # Replace with decreasing probability\n                j = torch.randint(0, self.total_seen, (1,)).item()\n                if j < self.replay_buffer_size:\n                    self.replay_buffer_x[j] = x[i].cpu()\n                    self.replay_buffer_y[j] = y[i].cpu()\n\n    def get_replay_batch(self, batch_size: int) -> tuple[torch.Tensor, torch.Tensor] | None:\n        \"\"\"Sample a batch from the replay buffer.\"\"\"\n        if len(self.replay_buffer_x) < batch_size:\n            return None\n        indices = torch.randint(0, len(self.replay_buffer_x), (batch_size,))\n        x = torch.stack([self.replay_buffer_x[i] for i in indices])\n        y = torch.stack([self.replay_buffer_y[i] for i in indices])\n        return x.to(self.device), y.to(self.device)\n\n    def incremental_update(\n        self,\n        new_dataloader: DataLoader,\n        epochs: int = 3,\n        replay_ratio: float = 0.3,  # 30% of each batch from replay\n    ):\n        \"\"\"Update model on new data with forgetting prevention.\n\n        Recipe:\n        1. Low learning rate with warmup\n        2. Mix new data with replay buffer samples\n        3. EWC penalty on loss\n        4. Few epochs (3-5) to avoid overfitting to new regime\n        \"\"\"\n        self.model.train()\n        criterion = nn.CrossEntropyLoss()\n\n        optimizer = optim.AdamW(\n            self.model.parameters(), lr=self.base_lr, weight_decay=1e-4\n        )\n\n        step = 0\n        for epoch in range(epochs):\n            for x, y in new_dataloader:\n                x, y = x.to(self.device), y.to(self.device)\n\n                # Learning rate warmup\n                if step < self.warmup_steps:\n                    lr = self.base_lr * (step + 1) / self.warmup_steps\n                    for pg in optimizer.param_groups:\n                        pg[\"lr\"] = lr\n\n                # Mix with replay buffer\n                replay = self.get_replay_batch(int(x.size(0) * replay_ratio))\n                if replay is not None:\n                    rx, ry = replay\n                    x = torch.cat([x, rx], dim=0)\n                    y = torch.cat([y, ry], dim=0)\n\n                # Forward\n                out = self.model(x)\n                loss = criterion(out[\"logits\"], y)\n\n                # Add EWC penalty\n                ewc_loss = self.ewc_penalty()\n                total_loss = loss + self.ewc_lambda * ewc_loss\n\n                optimizer.zero_grad(set_to_none=True)\n                total_loss.backward()\n                torch.nn.utils.clip_grad_norm_(self.model.parameters(), max_norm=1.0)\n                optimizer.step()\n\n                # Add new data to replay buffer\n                self.add_to_replay_buffer(x[:len(x) - (len(replay[0]) if replay else 0)], \n                                          y[:len(y) - (len(replay[1]) if replay else 0)])\n\n                step += 1\n\n            logger.info(\n                \"Incremental epoch %d: loss=%.4f ewc_penalty=%.4f buffer_size=%d\",\n                epoch, loss.item(), ewc_loss.item(), len(self.replay_buffer_x),\n            )\n\n        # Update Fisher information after adaptation\n        self.compute_fisher_information(new_dataloader)\n```\n\n---\n\n## 10. Ensemble of Neural Models\n\n```python\nimport torch\nimport torch.nn as nn\nimport numpy as np\nimport logging\nfrom typing import List\n\nlogger = logging.getLogger(__name__)\n\n\nclass DeepEnsemble:\n    \"\"\"Train N models with different random seeds. Use disagreement as uncertainty.\n\n    Deep ensembles (Lakshminarayanan et al., 2017) are the gold standard for\n    neural network uncertainty. For 5-min crypto:\n    - Train 5 GRUs with different seeds\n    - Prediction = average softmax probabilities\n    - Uncertainty = entropy of average prediction OR variance across models\n    - Only trade when uncertainty is low (models agree)\n\n    Expected results:\n    - Ensemble accuracy: +1-3% over single model\n    - High-confidence subset (top 50%): +5-8% accuracy improvement\n    - This filtering alone is worth the 5x training cost\n    \"\"\"\n\n    def __init__(\n        self,\n        model_class: type,\n        model_kwargs: dict,\n        n_models: int = 5,\n        device: torch.device = torch.device(\"cpu\"),\n    ):\n        self.n_models = n_models\n        self.device = device\n        self.models: List[nn.Module] = []\n\n        for i in range(n_models):\n            torch.manual_seed(42 + i * 1000)\n            np.random.seed(42 + i * 1000)\n            model = model_class(**model_kwargs).to(device)\n            self.models.append(model)\n            logger.info(\"Initialized ensemble member %d\", i)\n\n    def train_member(\n        self,\n        model_idx: int,\n        train_loader: DataLoader,\n        val_loader: DataLoader,\n        config: TrainingConfig,\n    ) -> dict:\n        \"\"\"Train a single ensemble member.\"\"\"\n        trainer = Trainer(self.models[model_idx], config, self.device)\n        history = trainer.fit(train_loader, val_loader)\n        return history\n\n    @torch.no_grad()\n    def predict(self, x: torch.Tensor) -> dict[str, torch.Tensor]:\n        \"\"\"Ensemble prediction with uncertainty estimation.\n\n        Returns:\n        - mean_probs: averaged softmax across ensemble\n        - predicted_class: argmax of mean_probs\n        - entropy: predictive entropy (uncertainty)\n        - disagreement: fraction of models that disagree with majority\n        -
