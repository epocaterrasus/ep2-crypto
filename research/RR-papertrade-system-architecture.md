Paper Trading with Full Position Management\n\n### Margin, Liquidation, and Funding Rate Simulation\n\n```python\n# position_simulator.py\nimport structlog\nfrom decimal import Decimal\nfrom datetime import datetime, timezone\n\nlogger = structlog.get_logger()\n\n\nclass PositionSimulator:\n    \"\"\"\n    Full position lifecycle simulation including:\n    - Margin calculation (cross and isolated)\n    - Liquidation price tracking\n    - Funding rate payments (8-hourly)\n    - Unrealized PnL mark-to-market\n    \"\"\"\n\n    # Binance-like maintenance margin rates (tiered)\n    MAINTENANCE_MARGIN_TIERS = [\n        # (max_notional, maintenance_margin_rate, maintenance_amount)\n        (50_000, Decimal(\"0.004\"), Decimal(\"0\")),\n        (250_000, Decimal(\"0.005\"), Decimal(\"50\")),\n        (1_000_000, Decimal(\"0.01\"), Decimal(\"1300\")),\n        (5_000_000, Decimal(\"0.025\"), Decimal(\"16300\")),\n        (20_000_000, Decimal(\"0.05\"), Decimal(\"141300\")),\n    ]\n\n    FUNDING_INTERVAL_HOURS = 8  # Binance: 00:00, 08:00, 16:00 UTC\n\n    def __init__(\n        self,\n        initial_balance: Decimal,\n        leverage: int = 10,\n        margin_mode: str = \"cross\",  # \"cross\" or \"isolated\"\n    ):\n        self._balance = initial_balance\n        self._leverage = leverage\n        self._margin_mode = margin_mode\n        self._positions: dict[str, dict] = {}\n        self._cumulative_funding: Decimal = Decimal(\"0\")\n        self._last_funding_time: Optional[datetime] = None\n        self._funding_history: list[dict] = []\n\n    def calculate_liquidation_price(\n        self,\n        entry_price: Decimal,\n        quantity: Decimal,\n        side: OrderSide,\n        leverage: int,\n    ) -> Decimal:\n        \"\"\"\n        Calculate liquidation price using Binance-like formula.\n\n        For LONG:\n          liq_price = entry_price * (1 - 1/leverage + maintenance_margin_rate)\n\n        For SHORT:\n          liq_price = entry_price * (1 + 1/leverage - maintenance_margin_rate)\n\n        This is simplified; real Binance uses wallet balance and\n        all positions for cross margin.\n        \"\"\"\n        notional = entry_price * quantity\n        mmr = self._get_maintenance_margin_rate(notional)\n\n        if side == OrderSide.BUY:\n            liq_price = entry_price * (1 - Decimal(\"1\") / leverage + mmr)\n        else:\n            liq_price = entry_price * (1 + Decimal(\"1\") / leverage - mmr)\n\n        return liq_price\n\n    def _get_maintenance_margin_rate(self, notional: Decimal) -> Decimal:\n        for max_notional, mmr, _ in self.MAINTENANCE_MARGIN_TIERS:\n            if notional <= max_notional:\n                return mmr\n        return self.MAINTENANCE_MARGIN_TIERS[-1][1]\n\n    def check_liquidation(self, symbol: str, current_price: Decimal) -> bool:\n        \"\"\"Check if current price would trigger liquidation.\"\"\"\n        pos = self._positions.get(symbol)\n        if pos is None:\n            return False\n\n        liq_price = pos[\"liquidation_price\"]\n        if pos[\"side\"] == OrderSide.BUY:\n            return current_price <= liq_price\n        else:\n            return current_price >= liq_price\n\n    async def apply_funding_rate(\n        self,\n        symbol: str,\n        funding_rate: Decimal,\n        mark_price: Decimal,\n        current_time: datetime,\n    ):\n        \"\"\"\n        Apply funding rate payment/receipt.\n        Called at each 8-hour interval.\n\n        Funding payment = position_quantity * mark_price * funding_rate\n        - Positive rate: longs pay shorts\n        - Negative rate: shorts pay longs\n        \"\"\"\n        pos = self._positions.get(symbol)\n        if pos is None:\n            return\n\n        notional = pos[\"quantity\"] * mark_price\n        funding_payment = notional * funding_rate\n\n        if pos[\"side\"] == OrderSide.BUY:\n            # Long pays when rate is positive\n            self._balance -= funding_payment\n        else:\n            # Short receives when rate is positive\n            self._balance += funding_payment\n\n        self._cumulative_funding += abs(funding_payment)\n\n        record = {\n            \"timestamp\": current_time.isoformat(),\n            \"symbol\": symbol,\n            \"funding_rate\": str(funding_rate),\n            \"mark_price\": str(mark_price),\n            \"position_side\": pos[\"side\"].value,\n            \"position_quantity\": str(pos[\"quantity\"]),\n            \"funding_payment\": str(funding_payment),\n            \"balance_after\": str(self._balance),\n        }\n        self._funding_history.append(record)\n\n        logger.info(\"funding_rate_applied\", **record)\n\n    def mark_to_market(self, symbol: str, current_price: Decimal) -> Decimal:\n        \"\"\"Calculate unrealized PnL at current mark price.\"\"\"\n        pos = self._positions.get(symbol)\n        if pos is None:\n            return Decimal(\"0\")\n\n        if pos[\"side\"] == OrderSide.BUY:\n            pnl = (current_price - pos[\"entry_price\"]) * pos[\"quantity\"]\n        else:\n            pnl = (pos[\"entry_price\"] - current_price) * pos[\"quantity\"]\n\n        pos[\"unrealized_pnl\"] = pnl\n        return pnl\n\n    def get_margin_utilization(self) -> Decimal:\n        \"\"\"\n        What fraction of available balance is used as margin.\n        Should stay below 30% for safety.\n        \"\"\"\n        total_margin = sum(\n            p[\"quantity\"] * p[\"entry_price\"] / self._leverage\n            for p in self._positions.values()\n        )\n        if self._balance <= 0:\n            return Decimal(\"1.0\")  # fully utilized or bankrupt\n        return total_margin / self._balance\n```\n\n---\n\n## 9.
