self._xgb_weight) * gru_pred\n\n        return {\n            \"prediction\": float(combined),\n            \"xgb_pred\": float(xgb_pred),\n            \"gru_pred\": float(gru_pred),\n            \"confidence\": float(1.0 - Graceful Degradation\n\n### Circuit Breaker Pattern\n\n```python\nimport time\nimport enum\nimport structlog\n\nlogger = structlog.get_logger()\n\nclass CircuitState(enum.Enum):\n    CLOSED = \"closed\"        # Normal operation\n    OPEN = \"open\"            # Failing, reject calls\n    HALF_OPEN = \"half_open\"  # Testing recovery\n\nclass CircuitBreaker:\n    def __init__(self, name: str, failure_threshold: int = 5,\n                 recovery_timeout: float = 30.0, half_open_max: int = 1):\n        self.name = name\n        self._state = CircuitState.CLOSED\n        self._failure_count = 0\n        self._failure_threshold = failure_threshold\n        self._recovery_timeout = recovery_timeout\n        self._half_open_max = half_open_max\n        self._last_failure_time = 0.0\n        self._half_open_calls = 0\n\n    @property\n    def state(self) -> CircuitState:\n        if self._state == CircuitState.OPEN:\n            if time.monotonic() - self._last_failure_time >= self._recovery_timeout:\n                self._state = CircuitState.HALF_OPEN\n                self._half_open_calls = 0\n                logger.info(\"circuit_half_open\", breaker=self.name)\n        return self._state\n\n    def record_success(self) -> None:\n        if self._state == CircuitState.HALF_OPEN:\n            self._state = CircuitState.CLOSED\n            logger.info(\"circuit_closed\", breaker=self.name)\n        self._failure_count = 0\n\n    def record_failure(self) -> None:\n        self._failure_count += 1\n        self._last_failure_time = time.monotonic()\n        if self._failure_count >= self._failure_threshold:\n            self._state = CircuitState.OPEN\n            logger.warning(\"circuit_opened\", breaker=self.name,\n                           failures=self._failure_count)\n\n    def allow_request(self) -> bool:\n        state = self.state\n        if state == CircuitState.CLOSED:\n            return True\n        if state == CircuitState.HALF_OPEN:\n            self._half_open_calls += 1\n            return self._half_open_calls <= self._half_open_max\n        return False  # OPEN\n```\n\n### Data Source Failover\n\n```python\nclass ResilientDataManager:\n    \"\"\"\n    Maintains predictions even when data sources partially fail.\n    Key principle: degrade gracefully, never silently produce garbage.\n    \"\"\"\n\n    def __init__(self):\n        self._source_status: dict[str, CircuitBreaker] = {\n            \"binance_ws\": CircuitBreaker(\"binance_ws\"),\n            \"bybit_ws\": CircuitBreaker(\"bybit_ws\"),\n            \"funding_rate\": CircuitBreaker(\"funding_rate\"),\n        }\n        self._stale_thresholds = {\n            \"price\": 5.0,        # seconds — price data stale after 5s\n            \"funding\": 300.0,    # seconds — funding rates update every 8h\n            \"orderbook\": 10.0,   # seconds\n        }\n        self._last_update: dict[str, float] = {}\n\n    def is_data_fresh(self, data_type: str) -> bool:\n        last = self._last_update.get(data_type, 0)\n        threshold = self._stale_thresholds.get(data_type, 30.0)\n        return (time.monotonic() - last) < threshold\n\n    def get_available_features(self) -> tuple[list[str], list[str]]:\n        \"\"\"Returns (available_features, missing_features).\"\"\"\n        available = []\n        missing = []\n\n        if self.is_data_fresh(\"price\"):\n            available.extend([\"returns\", \"volatility\", \"rsi\", \"bbands\"])\n        else:\n            missing.extend([\"returns\", \"volatility\", \"rsi\", \"bbands\"])\n\n        if self.is_data_fresh(\"funding\"):\n            available.append(\"funding_rate\")\n        else:\n            missing.append(\"funding_rate\")\n\n        return available, missing\n\n    def should_predict(self) -> tuple[bool, str]:\n        \"\"\"\n        Decision: can we make a reliable prediction?\n        Returns (should_predict, reason).\n        \"\"\"\n        available, missing = self.get_available_features()\n\n        # Must have price data — non-negotiable\n        if not self.is_data_fresh(\"price\"):\n            return False, \"price_data_stale\"\n\n        # If >30% of features missing, skip prediction\n        total = len(available) + len(missing)\n        if total > 0 and len(missing) / total > 0.3:\n            return False, f\"too_many_missing_features:{len(missing)}/{total}\"\n\n        return True, \"ok\"\n```\n\n---\n\n## 9. Memory Leak Prevention\n\n### Common Leaks in Long-Running Python\n\n| Source | How It Leaks | Fix |\n|---|---|---|\n| Unbounded lists/dicts | Appending without evicting | Use `deque(maxlen=N)` or ring buffers |\n| Callback/listener references | Objects retained by event handlers | Use `weakref`, explicit deregister |\n| Exception objects | Traceback holds frame locals | Don't store exceptions long-term |\n| Logging with large objects | Logger retains reference to args | Log primitive values, not objects |\n| pandas DataFrame copies | `.append()` creates copies | Pre-allocate, use `.iloc[]` |\n| asyncio Tasks not awaited | Task + its result sit in memory | Always gather or track tasks |\n| Third-party C extensions | Hidden internal caches | Monitor RSS, restart periodically |\n\n### Detection and Prevention\n\n```python\nimport gc\nimport sys\nimport tracemalloc\nimport asyncio\nimport structlog\n\nlogger = structlog.get_logger()\n\nclass MemoryWatchdog:\n    \"\"\"Periodic memory monitoring for long-running processes.\"\"\"\n\n    def __init__(self, warning_mb: float = 512, critical_mb: float = 1024,\n                 check_interval: float = 60.0):\n        self._warning_mb = warning_mb\n        self._critical_mb = critical_mb\n        self._interval = check_interval\n        self._baseline_mb: float | None = None\n\n    async def run(self) -> None:\n        import psutil\n        process = psutil.Process()\n\n        # Enable tracemalloc for debugging (small overhead ~3-5%)\n        tracemalloc.start(10)  # 10 frames deep\n\n        while True:\n            await asyncio.sleep(self._interval)\n\n            rss_mb = process.memory_info().rss / (1024 * 1024)\n\n            if self._baseline_mb is None:\n                self._baseline_mb = rss_mb\n\n            growth = rss_mb - self._baseline_mb\n\n            if rss_mb > self._critical_mb:\n                # Dump top allocators for debugging\n                snapshot = tracemalloc.take_snapshot()\n                top = snapshot.statistics(\"lineno\")[:10]\n                for stat in top:\n                    logger.error(\"memory_top_allocator\", stat=str(stat))\n\n                # Force GC collection\n                collected = gc.collect()\n                logger.error(\"memory_critical\",\n                             rss_mb=round(rss_mb, 1),\n                             growth_mb=round(growth, 1),\n                             gc_collected=collected)\n\n            elif rss_mb > self._warning_mb:\n                logger.warning(\"memory_warning\",\n                               rss_mb=round(rss_mb, 1),\n                               growth_mb=round(growth, 1))\n```\n\n### Practical Tips\n\n1. **Run `gc.collect()` periodically** (every 5-10 minutes) — Python's generational GC does not always collect cyclic references promptly.\n2. **Set `gc.set_threshold(700, 10, 5)`** — more aggressive collection for gen1/gen2 (defaults are 700, 10, 10).\n3. **Use `__slots__`** on hot-path dataclasses — reduces per-instance memory by 30-50%.\n4. **Scheduled restarts** are acceptable — restart every 24 hours via systemd or Docker restart policy. This is pragmatic, not lazy.\n\n---\n\n## 13.
